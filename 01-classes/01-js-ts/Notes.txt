alias - export 
-> file1.js
    const A = 10
    export {A as numberA}

-> file2.js
    import {numberA} from "./file1"


alias - import
-> file1.js
    const A = 10
    export {A}

-> file2.js
    import {A as numberA} from "./file1"


direct export
-> file1.js
    export const A = 10

-> file2.js
    import {A} from "./file1"


import all

    export const A = 10
    export const B = 0

-> file2.js
    import * as numbers from "./file1"

    const A = numbers.A
    const B = numbers.B


export function as DEFAULT

        return a + b
    }

export default variable/function
